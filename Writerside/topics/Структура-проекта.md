# 📂 Структура проекта

```bash
project/
├── build/              # Скомпилированный код
├── node_modules/       # Зависимости Node.js
├── src/                
│   ├── crons/                        # Крон-задачи (регулярные фоновые процессы)
│   │   ├── cron-handlers/            # Обработчики задач
│   │   │   ├── utils/                # Функции для кронов
│   │   │   │   ├── saveToMongoUnifiedProducts.ts       # Парсинг и запись всех товаров поставщиков в MongoDB
│   │   │   │   └── upsertProductsToSalesDrive.ts       # Добавление/обновление товаров из БД в SalesDrive
│   │   │   └── index.ts              # Экспорт хендлеров и их параметры и колбеки
│   │   ├── index.ts                  # Точка входа для крон-задач
│   │   └── jobs.ts                   # Регистрация заданий

│   ├── lib/                        # Общая библиотека проекта
│   │   ├── databases/              # Работа с базой данных (MongoDB)
│   │   ├── interfaces/             # TypeScript-интерфейсы и типы
│   │   ├── log/                    # Логирование
│   │   ├── salesdrive/             # Интеграция API SalesDrive
│   │   ├── suppliers/              # Логика парсинга поставщиков
│   │   └── utils/                  # Вспомогательные функции

│   ├── server/                        # API-сервер и роутинг
│   │   ├── handlers.ts              # Обработчики HTTP-запросов
│   │   └── index.ts                 # Инициализация сервера, подключение middleware и маршрутов

│   ├── index.ts        # Точка входа в приложение
│   └── test.ts         # Используется в режиме разработки
├── .editorconfig       # Общие правила форматирования кода
├── .env                # Переменные окружения
├── .eslintrc.json      # Конфигурация ESLint
├── .gitignore          # Файлы и папки, игнорируемые Git
├── .prettierrc.json    # Конфигурация Prettier
├── nodemon.json        # Настройки nodemon
├── nodemon-test.json   # Настройки nodemon для тестов
├── package.json        # Список зависимостей и npm-скрипты
├── package-lock.json   # Зафиксированные версии зависимостей
└── tsconfig.json       # Конфигурация TypeScript
